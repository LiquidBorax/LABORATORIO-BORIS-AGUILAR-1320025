using System;

class Program
{
    static void Main(string[] args)
    {
        int opcion;

        Console.WriteLine("==================================");
        Console.WriteLine("          MENU PRINCIPAL");
        Console.WriteLine("==================================");
        Console.WriteLine("1 - Clasificación de Vehículo");
        Console.WriteLine("2 - Aumento Límite de Crédito");
        Console.WriteLine("3 - Evaluación de Empleado");
        Console.WriteLine("4 - Pizzería Bella Napoli");
        Console.WriteLine("==================================");
        Console.Write("Seleccione una opción: ");
        opcion = int.Parse(Console.ReadLine());

        switch (opcion)
        {
            case 1:

                int tipo;

                Console.Write("Ingrese su tipo de vehiculo 1: Bicicleta, 2: Motocicleta, 3: Auto, 4: Camión, 5: Autobús: ");
                tipo = int.Parse(Console.ReadLine());

                switch (tipo)
                {
                    case 1:
                        Console.WriteLine("Bicicleta -> No motorizado");
                        break;
                    case 2:
                        Console.WriteLine("Motocicleta -> Ligero");
                        break;
                    case 3:
                        Console.WriteLine("Auto -> Mediano");
                        break;
                    case 4:
                        Console.WriteLine("Camión -> Pesado");
                        break;
                    case 5:
                        Console.WriteLine("Autobús -> Transporte público");
                        break;
                    default:
                        Console.WriteLine("Error: número fuera de rango.");
                        break;
                }

                break;

            case 2:

                int tipoTarjeta;
                double limite;
                double nuevoLimite = 0;

                Console.Write("Ingrese tipo de tarjeta: ");
                tipoTarjeta = int.Parse(Console.ReadLine());

                Console.Write("Ingrese límite actual: ");
                limite = double.Parse(Console.ReadLine());

                switch (tipoTarjeta)
                {
                    case 1:
                        nuevoLimite = limite + (limite * 0.25);
                        break;
                    case 2:
                        nuevoLimite = limite + (limite * 0.35);
                        break;
                    case 3:
                        nuevoLimite = limite + (limite * 0.40);
                        break;
                    default:
                        nuevoLimite = limite + (limite * 0.50);
                        break;
                }

                Console.WriteLine("Nuevo límite de crédito: " + nuevoLimite);

                break;

            case 3:

                double puntos;
                double dinero;

                Console.Write("Ingrese puntuación (0.0, 0.4, 0.6 o más): ");
                puntos = double.Parse(Console.ReadLine());

                if (puntos == 0.0)
                {
                    Console.WriteLine("Nivel: Inaceptable");
                }
                else if (puntos == 0.4)
                {
                    Console.WriteLine("Nivel: Aceptable");
                }
                else if (puntos >= 0.6)
                {
                    Console.WriteLine("Nivel: Meritorio");
                }
                else
                {
                    Console.WriteLine("Puntuación inválida.");
                }

                dinero = 2400 * puntos;

                Console.WriteLine("Dinero recibido: " + dinero);

                break;

            case 4:

                int tipoPizza;
                int ingrediente;

                Console.WriteLine("1 - Vegetariana");
                Console.WriteLine("2 - No Vegetariana");
                Console.Write("Seleccione tipo de pizza: ");
                tipoPizza = int.Parse(Console.ReadLine());

                switch (tipoPizza)
                {
                    case 1:

                        Console.WriteLine("Ingredientes disponibles:");
                        Console.WriteLine("1 - Pimiento");
                        Console.WriteLine("2 - Tofu");
                        ingrediente = int.Parse(Console.ReadLine());

                        Console.WriteLine("Pizza Vegetariana con mozzarella, tomate y:");

                        switch (ingrediente)
                        {
                            case 1:
                                Console.WriteLine("Pimiento");
                                break;
                            case 2:
                                Console.WriteLine("Tofu");
                                break;
                            default:
                                Console.WriteLine("Ingrediente inválido.");
                                break;
                        }

                        break;

                    case 2:

                        Console.WriteLine("Ingredientes disponibles:");
                        Console.WriteLine("1 - Peperoni");
                        Console.WriteLine("2 - Jamón");
                        Console.WriteLine("3 - Salmón");
                        ingrediente = int.Parse(Console.ReadLine());

                        Console.WriteLine("Pizza No Vegetariana con mozzarella, tomate y:");

                        switch (ingrediente)
                        {
                            case 1:
                                Console.WriteLine("Peperoni");
                                break;
                            case 2:
                                Console.WriteLine("Jamón");
                                break;
                            case 3:
                                Console.WriteLine("Salmón");
                                break;
                            default:
                                Console.WriteLine("Ingrediente inválido.");
                                break;
                        }

                        break;

                    default:
                        Console.WriteLine("Tipo de pizza inválido.");
                        break;
                }

                break;

            default:
                Console.WriteLine("Opción inválida.");
                break;
        }

        Console.ReadKey();
    }
}
